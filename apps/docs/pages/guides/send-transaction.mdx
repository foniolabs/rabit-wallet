# Send a transaction

The `useSendTransaction` hook returns a function and async state for sending an EVM transaction from the connected account.

```tsx filename="components/send.tsx"
'use client'
import { useSendTransaction } from '@rabit/react'
import { parseEther } from 'viem'

export function Send() {
  const { sendTransaction, isPending, data: hash } = useSendTransaction()

  return (
    <div>
      <button
        disabled={isPending}
        onClick={() =>
          sendTransaction({
            to: '0x000000000000000000000000000000000000dEaD',
            value: parseEther('0.001'),
          })
        }
      >
        {isPending ? 'Sending…' : 'Send 0.001 ETH'}
      </button>
      {hash && <p>tx: {hash}</p>}
    </div>
  )
}
```

## API

```ts
function useSendTransaction(): {
  sendTransaction: (args: { to: Address; value?: bigint; data?: Hex }) => Promise<Hex>
  isPending: boolean
  data: Hex | undefined
  error: Error | undefined
}
```
